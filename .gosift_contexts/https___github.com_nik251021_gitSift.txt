**Проект:** `goSift` (репозиторий `gitSift`)

**Назначение:** CLI-утилита на Go для автоматической генерации подробных, но сжатых технических справок (контекстов) по Git-репозиториям, используя Google Gemini API. Предназначена для быстрого обзора проекта и его ключевых характеристик.

**Стек Технологий:**
*   **Язык:** Go (1.26.2+)
*   **Зависимости:** `github.com/joho/godotenv` (для `.env`)
*   **Внешние сервисы:** Google Gemini API (`generativelanguage.googleapis.com`)

**Основные Команды CLI:**
1.  **`goSift sift <repoURL>`**: Главная команда. Клонирует репозиторий, собирает файлы, отправляет в Gemini API, сохраняет справку.
2.  **`goSift createContext <repoURL> <repoCode>`**: (Внутренняя/вспомогательная) Отправляет предоставленный код в Gemini API, сохраняет ответ.
3.  **`goSift setAPIkey <key>`**: Сохраняет API ключ Gemini в `.env`.
4.  **`goSift getAPIkey`**: Выводит текущий API ключ.
5.  **`goSift setCurrentLinkToRepo <url>`**: Сохраняет URL репозитория в `.env`.
6.  **`goSift getCurrentLinkToRepo`**: Выводит текущий URL репозитория.
7.  **`goSift help`**: Отображает справку.

**Рабочий Процесс `sift`:**
1.  Принимает URL репозитория.
2.  **`GetRepoFiles`**:
    *   Клонирует репозиторий (`git clone --depth 1`).
    *   Перечисляет файлы (`git ls-files`).
    *   Читает содержимое файлов (кроме `.env`, `.sum`).
    *   Формирует единую строку-слепок (`--- File: <path> ---\n```\n<content>\n```\n`).
3.  **`CreateContext`**:
    *   Получает `AGENT_API_KEY` из `.env`.
    *   Формирует промпт для Gemini с системной инструкцией и слепком репозитория.
    *   Выполняет HTTP POST запрос к Gemini API.
    *   Парсит JSON-ответ.
    *   Сохраняет текст AI-ответа в `.gosift_contexts/<safe_repo_filename>.txt`.

**Структура Данных/Конфигурация:**
*   `.env`: Хранит `AGENT_API_KEY`, `CURRENT_REPO_LINK`.
*   `.gosift_contexts/`: Директория для кэширования сгенерированных AI справок. Имена файлов формируются из URL репозитория.

**Особенности:**
*   Использование внешнего AI для автоматической интерпретации кода.
*   Локальное кэширование AI-ответов.
*   Минималистичный CLI для управления конфигурацией и запуска анализа.

// cmd/goSift/main.go - Точка входа и диспетчер команд
package main

import (
	"fmt"
	"os"

	"goSift/internal/handlers"
	"github.com/joho/godotenv"
)

type CommandHandler func(args []string) error

func main() {
	godotenv.Load() // Загрузка переменных окружения из .env

	commands := map[string]CommandHandler{
		"help":                 handlers.HelpHandler,
		"sift":                 handlers.SiftHandler, // Основная команда для анализа репозитория
		"createContext":        handlers.CreateContext, // Вызывается командой sift
		"setAPIkey":            handlers.SetApiKeyHandler,
		"getAPIkey":            handlers.GetApiKeyHandler,
		"setCurrentLinkToRepo": handlers.SetCurrentRepoHandler,
		"getCurrentLinkToRepo": handlers.GetCurrentRepoHandler,
	}

	if len(os.Args) < 2 {
		fmt.Println("Please, enter any arguments, or use goSift help")
		return
	}

	curcmd := os.Args[1]
	args := os.Args[2:]

	if handler, exists := commands[curcmd]; exists {
		err := handler(args)
		if err != nil {
			fmt.Println("Error in command", curcmd, err)
			return
		}
	} else {
		fmt.Println("Command not found", curcmd)
		return
	}
}
```

// internal/handlers/goSift.go - Обработчик основной команды 'sift'
package handlers

import (
	"fmt"
)

func SiftHandler(args []string) error {
	if len(args) < 1 || args[0] == "" {
		return fmt.Errorf("please provide a repository URL. Example: goSift sift <url>")
	}
	repoURL := args[0]

	fmt.Printf("Starting analysis for repository: %s\n", repoURL)

	// 1. Сборка файлов репозитория в единую строку.
	repoCode, err := GetRepoFiles(args) // args[0] это repoURL
	if err != nil {
		return fmt.Errorf("failed to collect repository files: %w", err)
	}

	// 2. Генерация и сохранение AI-контекста.
	fakeArgs := []string{repoURL, repoCode} // Формируем аргументы для CreateContext
	err = CreateContext(fakeArgs)
	if err != nil {
		return fmt.Errorf("failed to generate and save AI context: %w", err)
	}

	fmt.Println("\n🚀 [ALL DONE] project is ready!")
	return nil
}

```go
// internal/handlers/getRepoFiles.go - Клонирование и сбор файлов репозитория
package handlers

import (
	"fmt"
	"os"
	"os/exec"
	"strings"
)

func GetRepoFiles(args []string) (string, error) {
	if len(args) < 1 || args[0] == "" {
		return "", fmt.Errorf("no repository URL provided")
	}
	repoURL := args[0]

	tmpDir, err := os.MkdirTemp("", "gosift-") // Создание временной директории
	if err != nil {
		return "", fmt.Errorf("error creating temp dir: %w", err)
	}
	defer os.RemoveAll(tmpDir) // Удаление временной директории при выходе

	// Клонирование репозитория
	cloneCmd := exec.Command("git", "clone", "--depth", "1", repoURL, tmpDir)
	if err := cloneCmd.Run(); err != nil {
		return "", fmt.Errorf("error running git clone for %s: %w", repoURL, err)
	}
	
	// Получение списка файлов в репозитории
	listCmd := exec.Command("git", "ls-files")
	listCmd.Dir = tmpDir // Выполнение команды в директории клонированного репо
	out, err := listCmd.Output()
	if err != nil {
		return "", fmt.Errorf("error getting files: %w", err)
	}

	files := strings.Split(string(out), "\n")
	var builder strings.Builder
	builder.WriteString(fmt.Sprintf("=== string of repo: %s ===\n\n", repoURL))

	// Чтение содержимого каждого файла и формирование слепка
	for _, filePath := range files {
		filePath = strings.TrimSpace(filePath)
		if filePath == "" || filePath == ".env" || strings.HasSuffix(filePath, ".sum") {
			continue // Игнорирование пустых строк, .env и .sum файлов
		}

		fullPath := tmpDir + "/" + filePath
		content, err := os.ReadFile(fullPath)
		if err != nil {
			continue // Пропуск файлов, которые не удалось прочитать
		}

		builder.WriteString(fmt.Sprintf("--- File: %s ---\n", filePath))
		builder.WriteString("```\n")
		builder.Write(content)
		builder.WriteString("\n```\n\n")
	}

	return builder.String(), nil
}
